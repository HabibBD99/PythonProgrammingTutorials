   # 1. Class
   # 2. Object
   # 3. Inheritance
   # 4. Abstraction
   # 5.Encapsulation
   # 6.Polymorphism
   
   
   from abc import ABC, abstractmethod
   
   class Shape(ABC):
       def __init__(self,dim1,dim2):
             self.dim1 = dim1
             self.dim2 = dim2
   
       @abstractmethod
       def area(self):
          pass
   
   
   # Triangle
   class Triangle(Shape):
    def area(self):
        area = 0.5 * self.dim1 * self.dim2
        print("Area Of Triangle:", area)
   
   
   
   
   # Rectangle
   class Rectangle(Shape):
       def area(self):
           area = self.dim1 * self.dim2
           print("Area Of Rectangle:",area)
   
   r1 = Rectangle(20,30)
   r1.area()
   print("--------")
   
   
   t1 = Triangle(20,30)
   t1.area()

Python OOP Mazic Method


   # __init__(self,other)  (for Intialized)
   # __eq__(self,other)  (for ==)
   # __ne__(self,other)  (for !=)
   # __lt__(self,other)   (for <)
   # __gt__(self,other)   (for >)
   # __le__(self,other)   (for <=)
   # __ge__(self,other)   (for >=)
   # __add__(self,other)   (for +)
   # __mul__(self,other)   (for *)
   # __sub__(self,other)   (for -)
   # __div__(self,other)   (for /)
   # __str__(self,other)   (for "srting")
   
   
   
   class Bike:
       def __init__(self,name, color):
           self.name = name
           self.color = color
   
       def __eq__(self,other):
         return self.name == other.name and self.color == other.color
   
       def __str__(self):
           return (f"Name = {self.name},Color={self.color}")
   
       def display(self):
           print(f"Name = {self.name},Color={self.color}")
   
   
   bike1 = Bike("Yamaha R15","Blue")
   bike2 = Bike("Yamaha R15","Blue")
   print(bike1==bike2)

   if re.match(pettern,"colour Blue Is a  I Love Blue Colour"):

   if re.search(pettern, " Blue Is a colour I Love Blue Colour"):
       print("Match")
   else:
       print("Not Matched")
       print("----")
   
   
   
   
   pettern2 = r"col"
